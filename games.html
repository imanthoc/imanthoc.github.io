INCLUDE "hardware.inc"

SECTION "Header", ROM0[$100]
	jp entry_point
	ds $150 - @, 0 				; ds = define space

;@par de src
;@par hl dst
;@par bc len
memcopy:
	ld a, [de]		; a = *src
	ld [hl+], a		; *dst = a, dst++
	inc de			; src++
	dec bc			; len--

	ld a, b
	or a, c			; b or c
	jp nz, memcopy

	ret

update_keys:
	; Poll half the controller
	ld a, P1F_GET_BTN
	call .onenibble
	ld b, a ; B7-4 = 1; B3-0 = unpressed buttons

	; Poll the other half
	ld a, P1F_GET_DPAD
	call .onenibble
	swap a ; A7-4 = unpressed directions; A3-0 = 1
	xor a, b ; A = pressed buttons + directions
	ld b, a ; B = pressed buttons + directions

	; And release the controller
	ld a, P1F_GET_NONE
	ldh [rP1], a

	; Combine with previous cur_keys to make new_keys
	ld a, [cur_keys]
	xor a, b ; A = keys that changed state
	and a, b ; A = keys that changed to pressed
	ld [new_keys], a
	ld a, b
	ld [cur_keys], a
	ret

.onenibble
	ldh [rP1], a ; switch the key matrix
	call .knownret ; burn 10 cycles calling a known ret
	ldh a, [rP1] ; ignore value while waiting for the key matrix to settle
	ldh a, [rP1]
	ldh a, [rP1] ; this read counts
	or a, $F0 ; A7-4 = 1; A3-0 = unpressed keys
.knownret
	ret

wait_vblank:
	ld a, [rLY]
	cp 144
	jr c, wait_vblank
	ret

; borrowed from internet
random_step:
    ld b, a
    rlca             ; Rotate left
    bit 7, b
    jr z, .no_xor
    xor %00011101    ; Apply feedback taps (6,5,4,0)
.no_xor:
    ret

create_random_position:
loop_red_start:
	ld a, [random_seed]
	call random_step
	ld [random_seed], a

	cp a, 55
	jr nc, loop_red_start
loop_red_end:
	add a, 56
	ret

scroll_road:
	ld a, [rSCY]
	sub a, 4
	ld [rSCY], a
	ret

scroll_obstacle:
	ld a, [_OAMRAM + 4]
	add a, 4
	ld [_OAMRAM + 4], a
	ret

randomize_obstacle_position:
	ld a, [_OAMRAM + 4]

	cp a, 150
	jr c, end_if
if_true:
	call create_random_position
	ld [_OAMRAM + 5], a

	ld a, 0
	ld [_OAMRAM + 4], a
end_if:
	ret

check_collision:
	ld a, [_OAMRAM + 1]
	add a, 8				; a = car_x_end

	ld b, a					; b = car_x_end
	sub a, 8				; a = car_x_start
	ld d, a					; d = car_x_start

	ld a, [_OAMRAM + 5]
	add a, 4				; c = obstacle_x_center
	ld c, a
	; d = car_x_start, b = car_x_end, c = obs_x_center

	ld a, d
	cp c					; compare car_start and obs_center
	jr nc, no_col			; if car_x_start > obs_x_center

	ld a, b
	cp c					; compare car_end and obs_center
	jr c, no_col			; if car_x_end < obs_center
col_x:
	 ld a, [_OAMRAM]		; a = car_y_start
	 ld b, a				; b = car_y_start
	 add a, 16
	 ld c, a				; c = car_y_end
	 ld a, [_OAMRAM + 4]
	 add a, 4
	 ld d, a				; d = obs_y_center

	 ld a, b				; A = car_y_start
	 cp d					; compare car_y_start with obs_y_center
	 jr nc, no_col			; if car_y_start > obs_y_center

	 ld a, c				; A = car_y_end
	 cp d					; compare car_y_end with obs_y_center
	 jr c, no_col			; if car_y_end < obs_y_center
col_y:
	jp col_y

no_col:

	ret
move_car:
check_up:
	ld a, [cur_keys]
	and a, PADF_UP
	jp z, check_down
up:
	ld a, [_OAMRAM]
	dec a
	cp a, 17
	jp z, check_down
	ld [_OAMRAM], a

check_down:
	ld a, [cur_keys]
	and a, PADF_DOWN
	jp z, check_left
down:
	ld a, [_OAMRAM]
	inc a
	cp a, 143
	jp z, check_left
	ld [_OAMRAM], a

check_left:
	ld a, [cur_keys]
	and a, PADF_LEFT
	jp z, check_right
left:
	ld a, [_OAMRAM + 1]
	dec a
	cp a, 56
	jp z, check_right
	ld [_OAMRAM + 1], a

check_right:
	ld a, [cur_keys]
	and a, PADF_RIGHT
	jp z, exit
right:
	ld a, [_OAMRAM + 1]
	inc a
	cp a, 112
	jp z, exit
	ld [_OAMRAM + 1], a
exit:
	ret

entry_point:
	call wait_vblank

	ld a, 0
	ld [rLCDC], a

; copy tiles
	ld de, tiles
	ld hl, $9000
	ld bc, tiles_end - tiles
	call memcopy

; copy tilemap
	ld de, tilemap
	ld hl, $9800
	ld bc, tilemap_end - tilemap
	call memcopy

; copy car
	ld de, car
	ld hl, $8000
	ld bc, car_end - car
	call memcopy
; copy obstacle
	ld de, obstacle
	ld hl, $8020
	ld bc, obstacle_end - obstacle
	call memcopy

; clear oam
	ld a, 0
	ld hl, _OAMRAM
	ld b, 160
clear_oamram:
	ld [hl+], a

	dec b
	jp nz, clear_oamram

	ld hl, _OAMRAM
; load car object
	ld a, 120 + 16
	ld [hl+], a
	ld a, 64 + 8
	ld [hl+], a
	ld a, 0
	ld [hl+], a
	ld [hl+], a
; load obstacle object
	ld hl, _OAMRAM
	ld de, 4
	add hl, de
	ld a, 30 + 16
	ld [hl+], a
	ld a, 64 + 8
	ld [hl+], a
	ld a, 2
	ld [hl+], a
	ld a, 0
	ld [hl+], a
; turn lcd on
	ld a, LCDCF_ON | LCDCF_BGON | LCDCF_OBJON | LCDCF_OBJ16
    ld [rLCDC], a
	ld a, %11100100
	ld [rBGP], a
	ld a, %11100100
	ld [rOBP0], a

; initialize global variables.
	ld a, 0
	ld [frame_counter], a
	ld [cur_keys], a
	ld [new_keys], a

	ld a, [$FF04]
	ld [random_seed], a
main_loop:
	ld a, [rLY]
	cp 144
	jp nc, main_loop

	call wait_vblank

	call update_keys

	call move_car
	call scroll_road
	call scroll_obstacle
	call randomize_obstacle_position
	call check_collision

	jp main_loop
